var jsonData = pm.response.json();

pm.test("Response code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Department creation failure message is returned", function () {
    pm.expect(jsonData.status).to.eql("failed");
    pm.expect(jsonData.error).to.be.a("string").and.to.include("Department with such name exists");
});
