var jsonData = pm.response.json();

pm.test("Response status code is 403", function () {
    pm.response.to.have.status(403);
});

pm.test("Access denied message is correct", function () {
    pm.expect(jsonData.message).to.eql("Access denied: Lab Technician or Radiologist only");
});

    pm.expect(jsonData.user.role).to.be.a("string");
});
