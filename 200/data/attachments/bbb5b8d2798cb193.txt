var jsonData = pm.response.json();

pm.test("Status should be failed", function () {
    pm.expect(jsonData.status).to.eql("failed");
});

pm.test("Error array should exist", function () {
    pm.expect(jsonData).to.have.property("error").that.is.an("array").with.lengthOf(1);
});

pm.test("Error should contain correct details", function () {
    var error = jsonData.error[0];
    pm.expect(error.type).to.eql("field");
    pm.expect(error.msg).to.eql("must be a valid email");
    pm.expect(error.path).to.eql("email");
    pm.expect(error.location).to.eql("body");
});
