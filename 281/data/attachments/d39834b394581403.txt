var jsonData = pm.response.json();

pm.test("Response code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Response status is failed", function () {
    pm.expect(jsonData.status).to.eql("failed");
});

pm.test("Error message is correct", function () {
    pm.expect(jsonData.error[0].msg).to.eql("Test name is required");
});

pm.test("Error path is correct", function () {
    pm.expect(jsonData.error[0].path).to.eql("testDetails.testName");
});
