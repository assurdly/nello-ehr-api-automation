var jsonData = pm.response.json();

pm.test("Response status is success", function () {
    pm.expect(jsonData.status).to.eql("success");
});

pm.test("Response code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Validate response body fields", function () {
    pm.expect(jsonData.result.testDetails.testName).to.not.be.null.and.to.be.a('string').that.is.not.empty;
    pm.expect(jsonData.result.testDetails.sampleType).to.not.be.null.and.to.be.a('string').that.is.not.empty;
    pm.expect(jsonData.result.testDetails.description).to.be.a('string').that.is.not.null;  // Null value expected
    pm.expect(jsonData.result.results.resultFile).to.not.be.null.and.to.be.a('string').that.is.not.empty;
    pm.expect(jsonData.result.results.notes).to.not.be.null.and.to.be.a('string').that.is.not.empty;
    pm.expect(jsonData._id).to.not.be.null.and.to.be.a('string').that.is.not.empty;
    pm.expect(jsonData.encounterUuid).to.not.be.null.and.to.be.a('string').that.is.not.empty;
    pm.expect(jsonData.userType).to.not.be.null.and.to.be.a('string').that.is.not.empty;
    pm.expect(jsonData.testType).to.not.be.null.and.to.be.a('string').that.is.not.empty;
    pm.expect(jsonData.status).to.not.be.null.and.to.be.a('string').that.is.not.empty;
    pm.expect(jsonData.uploadedBy).to.not.be.null.and.to.be.a('string').that.is.not.empty;
    pm.expect(jsonData.createdAt).to.not.be.null.and.to.be.a('string').that.is.not.empty;
    pm.expect(jsonData.updatedAt).to.not.be.null.and.to.be.a('string').that.is.not.empty;
    pm.expect(jsonData.__v).to.not.be.null.and.to.be.a('number').that.is.not.null;
});
