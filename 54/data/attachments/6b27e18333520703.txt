var jsonData = pm.response.json();

pm.test("Response status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response indicates success", function () {
    pm.expect(jsonData.status).to.eql("success");
    pm.expect(jsonData.sponsors).to.be.an("array").and.to.have.length.above(0);
});

pm.test("Sponsor details are valid", function () {
    var sponsor = jsonData.sponsors[0];

    pm.expect(sponsor._id).to.be.a("string").and.to.not.be.null;
    pm.expect(sponsor.name).to.be.a("string").and.to.not.be.null;
    pm.expect(sponsor.type).to.be.a("string").and.to.not.be.null;
    pm.expect(sponsor.uuid).to.be.a("string").and.to.not.be.null;
    pm.expect(sponsor.is_Active).to.be.a("boolean").and.to.not.be.null;
    pm.expect(sponsor.createdAt).to.be.a("string").and.to.not.be.null;
    pm.expect(sponsor.updatedAt).to.be.a("string").and.to.not.be.null;

});
