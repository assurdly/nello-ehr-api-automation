var jsonData = pm.response.json();

pm.test("Response has status code 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response contains expected fields", function () {
    pm.expect(jsonData).to.have.property("status").that.is.a("string");
    pm.expect(jsonData).to.have.property("message").that.is.a("string");
});

pm.test("Response status is success", function () {
    pm.expect(jsonData.status).to.eql("success");
});

pm.test("Success message is correct", function () {
    pm.expect(jsonData.message).to.eql("Drug Generic Name Added successfully");
});
