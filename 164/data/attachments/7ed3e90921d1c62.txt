var jsonData = pm.response.json();

pm.test("Response status is failed", function () {
    pm.expect(jsonData.status).to.eql("failed");
});

pm.test("Response code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Response contains an error array", function () {
    pm.expect(jsonData.error).to.be.an("array").and.to.have.length.above(0);
});

pm.test("Error path is email", function () {
    pm.expect(jsonData.error[0].path).to.eql("email");
});

pm.test("Error message specifies invalid email format", function () {
    pm.expect(jsonData.error[0].msg).to.eql("must be a valid email");
});
