var jsonData = pm.response.json();

pm.test("Response code is 404", () => {
    pm.response.to.have.status(404);
});

pm.test("Response status is failed", function () {
    pm.expect(jsonData.status).to.eql("failed");
});

pm.test("Office not found error message is correct", function () {
    pm.expect(jsonData.error).to.eql("office not found");
});
