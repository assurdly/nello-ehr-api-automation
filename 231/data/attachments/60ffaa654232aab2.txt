var jsonData = pm.response.json();

pm.test("Response code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response status is success", function () {
    pm.expect(jsonData.status).to.eql("success");
});

pm.test("Validate response fields", function () {
    pm.expect(jsonData.results._id).to.be.a("string");
    pm.expect(jsonData.results.encounterUuid).to.be.a("string");
    pm.expect(jsonData.results.userType).to.be.a('string');
    pm.expect(jsonData.results.testType).to.be.a('string');
    pm.expect(jsonData.results.status).to.eql('completed');
    pm.expect(jsonData.resulst.uploadedBy).to.be.a('string');
});
