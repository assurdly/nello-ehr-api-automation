var jsonData = pm.response.json();

pm.test("Response status code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Response contains validation errors", function () {
    pm.expect(jsonData.errors).to.be.an("array").and.to.have.length.above(0);
});

pm.test("Error message for price field is correct", function () {
    pm.expect(jsonData.errors[0]).to.be.an("object").and.to.not.be.null;
    pm.expect(jsonData.errors[0].path).to.eql("price");
    pm.expect(jsonData.errors[0].msg).to.eql("Price must be a number");
});
