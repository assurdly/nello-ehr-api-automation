var jsonData = pm.response.json();

pm.test("Response status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response contains imaging data", function () {
    pm.expect(jsonData.status).to.eql("success");
    pm.expect(jsonData.imaging).to.be.an("array").and.to.have.length.above(0);
});

pm.test("Imaging details are valid", function () {
    var imaging = jsonData.imaging[0];

    pm.expect(imaging.is_Active).to.be.a("boolean");
    pm.expect(imaging._id).to.be.a("string").and.to.not.be.null;
    pm.expect(imaging.type).to.be.a("string").and.to.not.be.null;
    pm.expect(imaging.name).to.be.a("string").and.to.not.be.null;
    pm.expect(imaging.price).to.be.a("number").and.to.not.be.null;
    pm.expect(imaging.sponsor_uuid).to.be.a("string").and.to.not.be.null;
    pm.expect(imaging.plan_code).to.be.a("string").and.to.not.be.null;
});
