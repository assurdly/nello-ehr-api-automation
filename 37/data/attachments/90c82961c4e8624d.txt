var jsonData = pm.response.json();

pm.test("Response has status code 401", function () {
    pm.response.to.have.status(401);
});

pm.test("Response contains expected field", function () {
    pm.expect(jsonData).to.have.property("message").that.is.a("string");
});

pm.test("Error meesage is correct", function () {
    pm.expect(jsonData.message).to.eql("Token is not valid");
});
