var jsonData = pm.response.json();

pm.test("Response code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Token validation is successful", function () {
    pm.expect(jsonData.valid).to.be.true;
    pm.expect(jsonData.user).to.be.an("object");
    pm.expect(jsonData.user.email).to.be.a("string").and.to.include("@");
    pm.expect(jsonData.user.iat).to.be.a("number").and.to.be.below(jsonData.user.exp);
});
