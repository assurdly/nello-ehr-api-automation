var jsonData = pm.response.json();

pm.test("Response status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response contains other service data", function () {
    pm.expect(jsonData.status).to.eql("success");
    pm.expect(jsonData.otherservice).to.be.an("array").and.to.have.length.above(0);
});

pm.test("Other service details are valid", function () {
    var service = jsonData.otherservice[0];

    pm.expect(service.is_Active).to.be.a("boolean");
    pm.expect(service._id).to.be.a("string").and.to.not.be.null;
    pm.expect(service.type).to.be.a("string").and.to.not.be.null;
    pm.expect(service.name).to.be.a("string").and.to.not.be.null;
    pm.expect(service.price).to.be.a("number").and.to.not.be.null;
    pm.expect(service.sponsor_uuid).to.be.a("string").and.to.not.be.null;
    pm.expect(service.plan_code).to.be.a("string").and.to.not.be.null;
});
