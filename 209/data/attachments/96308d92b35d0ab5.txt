var jsonData = pm.response.json();

pm.test("Response status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Roles data is valid", function () {
    pm.expect(jsonData.status).to.eql("success");
    pm.expect(jsonData.roles).to.be.an("array").and.to.have.length.above(0);
});

pm.test("First role contains required fields", function () {
    pm.expect(jsonData.roles[0]).to.be.an("object").and.to.not.be.null;
    pm.expect(jsonData.roles[0]._id).to.be.a("string").and.to.not.be.null;
    pm.expect(jsonData.roles[0].office_uuid).to.be.a("string").and.to.not.be.null;
    pm.expect(jsonData.roles[0].name).to.be.a("string").and.to.not.be.null;
    pm.expect(jsonData.roles[0].createdAt).to.be.a("string").and.to.not.be.null;
});
