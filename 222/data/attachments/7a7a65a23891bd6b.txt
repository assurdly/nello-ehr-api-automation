var jsonData = pm.response.json();

pm.test("Response code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response status is success", function () {
    pm.expect(jsonData.status).to.eql("success");
});

pm.test("Validate response fields", function () {
    pm.expect(jsonData.result.notes).to.be.a('string');
    pm.expect(jsonData.result._id).to.be.a("string");
    pm.expect(jsonData.result.encounterUuid).to.be.a("string");
    pm.expect(jsonData.result.userType).to.be.a('string');
    pm.expect(jsonData.result.testType).to.be.a('string');
    pm.expect(jsonData.result.status).to.eql('completed');
    pm.expect(jsonData.result.uploadedBy).to.be.a('string');
});
