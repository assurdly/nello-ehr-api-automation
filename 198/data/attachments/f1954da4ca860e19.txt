var jsonData = pm.response.json();
pm.test("Response status is failed", function () {
    pm.expect(jsonData.status).to.eql("failed");
});

pm.test("Response contains an error array", function () {
    pm.expect(jsonData.error).to.be.an("array").and.to.have.length.above(0);
});

pm.test("Error path is token", function () {
    pm.expect(jsonData.error[0].path).to.eql("token");
});

pm.test("Error message should specify that token is required", function () {
    pm.expect(jsonData.error[0].msg).to.eql("Token is required");
});
