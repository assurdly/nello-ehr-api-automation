var jsonData = pm.response.json();

pm.test("Response code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Department validation failed", function () {
    pm.expect(jsonData.status).to.eql("failed");
    pm.expect(jsonData.error).to.be.an("array").and.to.have.length.above(0);
    pm.expect(jsonData.error[0].msg).to.be.a("string");
    pm.expect(jsonData.error[1].msg).to.be.a("string");
});

pm.test("An error about the required Department name is returned", function () {
    pm.expect(jsonData.error[0].msg).to.eql("Department name is required");
});
