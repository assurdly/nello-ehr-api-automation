var jsonData = pm.response.json();

pm.test("Response status code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Response body contains failure status and password error details", function () {
    pm.expect(jsonData.status).to.eql("failed");
    pm.expect(jsonData.error).to.be.an("array").that.is.not.empty;
    pm.expect(jsonData.error[0].type).to.eql("field");
    pm.expect(jsonData.error[0].value).to.eql("");
    pm.expect(jsonData.error[0].path).to.eql("password");
    pm.expect(jsonData.error[0].location).to.eql("body");
});

pm.test("Error message about the required password is returned", function () {
    pm.expect(jsonData.error[0].msg).to.eql("password is required");
});
