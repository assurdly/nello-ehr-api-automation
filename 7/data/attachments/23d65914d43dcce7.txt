var jsonData = pm.response.json();

pm.test("Response code is 200", () => {
    pm.response.to.have.status(200);
});

pm.test("Departments data is valid", () => {
    pm.expect(jsonData.status).to.eql("success");
    pm.expect(jsonData.departments).to.be.an("array").and.to.have.length.above(0);
    pm.expect(jsonData.departments[0].name).to.be.a("string").and.to.not.be.null;;
    pm.expect(jsonData.departments[0].isActive).to.be.true;
});
