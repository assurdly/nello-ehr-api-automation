var jsonData = pm.response.json();

pm.test("Response status code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Response object is returned failure", function () {
    pm.expect(jsonData.status).to.eql("failed");
});

pm.test("Error message about the existing email is returned", function () {
    pm.expect(jsonData.error).to.be.a("string").and.to.eql("user with email already exists");
});
